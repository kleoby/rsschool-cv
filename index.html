// return masked string
function maskify(cc) {
   cc = cc.split('');
   for (var i = 0; i < 4; i++) {
       cc[i] = '#';
   }
   cc = cc.join('');
   return cc
}

function arithmetic(a, b, operator) {
    switch (operator) {
        case 'add':
            return a + b;
        case 'subtract':
            return a - b;
        case 'multiply':
            return a * b;
        case 'divide':
            return a / b;
    }
}
